=== "Configuration"
    
    **r1:**
    ```
    hostname r1
    buggy
    !
    logging file debug ../binTmp/zzz10r1-log.run
    !
    crypto ipsec ips
     group 02
     cipher aes256cbc
     hash sha1
     key $v10$dGVzdGVy
     exit
    !
    vrf definition tester
     exit
    !
    vrf definition v1
     rd 1:1
     exit
    !
    interface ethernet1
     vrf forwarding v1
     ipv4 address 9.9.9.1 255.255.255.252
     ipv6 address 9999::1 ffff::
     no shutdown
     no log-link-change
     exit
    !
    interface tunnel1
     tunnel vrf v1
     tunnel source ethernet1
     tunnel destination 9999::2
     tunnel mode gre
     macsec ips
     vrf forwarding v1
     ipv4 address 1.1.1.1 255.255.255.0
     ipv6 address 1234::1 ffff::
     no shutdown
     no log-link-change
     exit
    !
    !

    end
    ```
